\section{Model and parameters}
The model under investigation is
\begin{align}
	x_{n+1} &= (1-\varepsilon)r_1 x_n(1 - x_n) + \varepsilon r_2 y_n(1-y_n)\\
	y_{n+1} &= (1-\varepsilon)r_2 y_n(1 - y_n) + \varepsilon r_1 x_n(1-x_n)
\end{align}
which is equivalent to a non-linear perturbation of a linear map:

\begin{equation}
	\symbf{x}_{n+1} = \symbf{F}(\symbf{x}_n) = A \symbf{x}_n + \symbf{f}(\symbf{x}_n)
\end{equation}


with 
\begin{equation}
	A = 
    \begin{bmatrix}
			(1-\varepsilon)r_1 & \varepsilon r_2\\
			\varepsilon r_1 & (1- \varepsilon)r_2
    \end{bmatrix},
		\hspace{5mm}
		\symbf{f}(\symbf{x}) = A \symbf{g}(\symbf{x}),
		\hspace{5mm}
	\symbf{g}(\symbf{x}) = -x^2 \symbf{e}_x - y^2 \symbf{e}_y
		\hspace{5mm}
	\symbf{x} = \symbf{x}_n 
\end{equation}
so 

\begin{equation}
\symbf{F}(\symbf{x}) =
		A \symbf{x} +
		\symbf{f}(\symbf{x})
		=
		\begin{bmatrix}
			(1-\varepsilon)r_1 & \varepsilon r_2\\
			\varepsilon r_1 & (1- \varepsilon)r_2
    \end{bmatrix}
		\begin{bmatrix}
			x\\
			y
    \end{bmatrix}
		+
		\begin{bmatrix}
			(1-\varepsilon)r_1 & \varepsilon r_2\\
			\varepsilon r_1 & (1- \varepsilon)r_2
    \end{bmatrix}
		\begin{bmatrix}
			-x^2\\
			-y^2
    \end{bmatrix}.
\end{equation}

Looking for fixed points amounts to the root finding problem

\begin{align}
	\begin{split}
	\symbf{G}(\symbf{x}) &= \symbf{F}(\symbf{x}) - \symbf{x} \\ 
						 &= A \symbf{x} +\symbf{f}(\symbf{x}) - \symbf{x}\\
						 &= A \symbf{x} + A \symbf{g}(\symbf{x}) -\symbf{x}\\
						 &= A(\symbf{x} + \symbf{g}(\symbf{x})) - \symbf{x} \\
						 &= 0
	\end{split}\\
	\Rightarrow \symbf{x} + \symbf{g}(\symbf{x}) &= A^{-1}\symbf{x}\\
	\iff \symbf{g}(\symbf{x}) &= (A^{-1} - I)\symbf{x}\\
%\end{align*}\begin{align}
	A^{-1} &= \frac{1}{\det(A)}{\textrm{adj}}(A) =
	\frac{1}{\det(A)}
	\begin{bmatrix}
		(1-\varepsilon)r_2 & -\varepsilon r_1\\
		-\varepsilon r_2 & (1-\varepsilon) r_1
	\end{bmatrix}^T
	\label{eq:singularityeq}
	\\
	\det(A) &= (1-\varepsilon)^2 r_1 r_2 - \varepsilon^2 r_1 r_2 = (1 - 2\varepsilon)r_1 r_2 
	\\
	\iff 
	\begin{bmatrix}
		-x^2\\
		-y^2
	\end{bmatrix}
	&=
	\begin{bmatrix}
		a & b\\
		c & d
	\end{bmatrix}
	\begin{bmatrix}
		x\\
		y
	\end{bmatrix}
\end{align}

\begin{equation}
	\begin{cases}
		a = \frac{1-\varepsilon}{r_1(1-2\varepsilon)}-1 = \frac{\varepsilon - 1}{2r_1(\varepsilon - 1/2)}-1\\
		b = \frac{-\varepsilon}{r_1(1-2\varepsilon)} = \frac{\varepsilon}{2r_1(\varepsilon - 1/2)}\\
		c = \frac{-\varepsilon}{r_2(1-2\varepsilon)} = \frac{\varepsilon}{2r_2(\varepsilon - 1/2)}\\
		d = \frac{1-\varepsilon}{r_2(1-2\varepsilon)} -1 = \frac{\varepsilon - 1}{2r_2(\varepsilon - 1/2)}-1
	\end{cases}
	\label{eq:system-help-vars}
\end{equation}

\begin{equation}
	\begin{cases}
		x^2 + ax + by = x(x+a) + by = 0\\
		y^2 + cx + dy = cx + y(y+d) = 0
	\end{cases}
	\label{eq:system}
\end{equation}

Thus the fixed points belong to the intersections of 2 parabolic surfaces. The simple nullcline (that is if we ought to find a fixed point it 
is necessary for it to lie on this curve) is given by the first equation above
\begin{equation}
	y(x) = \frac{-x^2-ax}{b}
	\label{eq:nullcline}
\end{equation}
Inserting Eq (\ref{eq:nullcline}) in the lower equation in (\ref{eq:system}) gives the following quartic polynomial:
\begin{equation}
	p(x) = x^4 + 2a x^3 + (a^2 - bd)x^2 + b(cd - ad)x
\end{equation}
From this we can readily factor $x$ and thus $\symbf{x}_1 = (0, 0)$ is the trivial root. This leaves us with
\begin{equation}
	q(x) = x^3 + 2a x^2 + (a^2 - bd)x + bd(c - a)
\end{equation}
and the condition $q(x)=0$ as the relevant requirement for a fixed point.
Since $\deg(q) = 3$ and $y(x)$ is a one-to-one mapping for $b\neq 0$, we can have, at most, four fixed points. In our case, we might have even less
than four fixed points since we also treat population rates on the square $[0,1]^2$.

\subsection{Population equilibrium \& Fixed points}

\subsubsection{Symmetric fixed points}

If we consider the line parameterization $x = y = t$ where $t\in [0,1]$, which is the linear subspace of population rates of equal
magnitude, we could ask if there is a rate $t^*$ such that $\symbf{G}(t^*) = 0$? Let $x = y = t$, then

\begin{align}
	\symbf{G}(\symbf{x}) &= A\symbf{x} + A \symbf{g}(\symbf{x}) - \symbf{x} = 0\\
	&= A\symbf{x} + A \symbf{g}(\symbf{x}) - \symbf{x}\\
	&= \symbf{x} + \symbf{g}(\symbf{x}) - A^{-1}\symbf{x}\\
	&= \symbf{g}(\symbf{x}) + \symbf{x} - A^{-1}\symbf{x}\\
	&= -t^2
	\begin{bmatrix}
		1\\
		1
	\end{bmatrix}
	+
	t
	\begin{bmatrix}
		1\\1
	\end{bmatrix}
	-
	\frac{1}{(1-2\varepsilon)r_1r_2}
	\begin{bmatrix}
		(1-\varepsilon)r_2 & -\varepsilon r_2\\
		-\varepsilon r_1 & (1-\varepsilon) r_1
	\end{bmatrix}
	\begin{bmatrix}
		1\\1
	\end{bmatrix}
	\\
		&=
	-t^2
	\begin{bmatrix}
		1\\1
	\end{bmatrix}
	+t 
	\begin{bmatrix}
		1 - \frac{1}{r_1}\\1 - \frac{1}{r_2}
	\end{bmatrix}
	=
	\begin{bmatrix}
		0\\0
	\end{bmatrix}.
\end{align}

If we disregard the trivial fixed point $t\neq 0$, then we get that
\begin{equation}
	\begin{cases}
		t = 1 - \frac{1}{r_1}\\
		t = 1 - \frac{1}{r_2}\\	
	\end{cases}
\end{equation}
which can hold simultaneously only if $r_1 = r_2 = r$. Thus, except for the trivial fixed point, we can have a fixed point of equally 
large population rates if the growth factors of both populations are equal. The second fixed point in the diagonal case is thus along a
line through the origin $\symbf{x}=\lambda(r)\symbf{v}$: 

\begin{equation}
	\symbf{x}_1(r) = \left(1 -\frac{1}{r}\right)
	\begin{bmatrix}
		1\\1
	\end{bmatrix}
\end{equation}
with the constraint $1\leq r$ since $(x, y) \in [0,1]^2$. The nontrivial diagonal fixed points form a one-parameter family
parameterized by $1\leq r$, corresponding to a one-dimensional branch of equilibria. Now we could linearize $\symbf{G}$ around
$\symbf{x}_1$ to see whether $\symbf{x}_1$ is an isolated root. We do this by computing the Jacobian (for $x = y = t$ and $r_1 = r_2 = r$):

\begin{align}
	\frac{\partial \symbf{G}}{\partial \symbf{x}} &= \frac{\partial \symbf{F}}{\partial \symbf{x}} - \frac{\partial \symbf{x}}{\partial \symbf{x}}\\
																								&= \frac{\partial \symbf{F}}{\partial \symbf{x}} - I\\
																								&= \begin{bmatrix}(1-\varepsilon)r(1-2t)&\varepsilon r(1 - 2t)\\
																									\varepsilon r(1 - 2t) & (1-\varepsilon)r(1-2t)
																									\end{bmatrix} - I\\
																								&=r(1-2t) 
																								\begin{bmatrix}(1-\varepsilon) & \varepsilon\\
																									\varepsilon & (1- \varepsilon)
																								\end{bmatrix} - I\\
																																							 &= (2-r)\begin{bmatrix}(1-\varepsilon) & \varepsilon\\
																									\varepsilon & (1- \varepsilon)
																								\end{bmatrix} - I\\
																									\lambda ^2 - {\rm tr}\left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right)\lambda + \det \left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right) &= 0\\
																									\lambda_{\pm} &=  \frac{1}{2}{\rm tr}\left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right) \pm \sqrt{ \frac{1}{4}{\rm tr}\left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right)^2 -  {\det}\left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right)}\\
																																&= (2 - r)(1-\varepsilon) - 1 \pm \varepsilon (2 - r)\\
																									\lambda_+&= 1-r\\
																									\lambda_-&=(2-r)(1-2\varepsilon) -1
\end{align}

Since $ {\det}\left(\frac{\partial \symbf{G}}{\partial \symbf{x}}\right) = \lambda_+\lambda_-$, we know by the inverse function theorem
that $\symbf{x}_1$ is an isolated zero to $\symbf{G}$ as long as neither $\lambda_+ = 0$ or $\lambda_- = 0$, otherwise the 
linearization becomes inconclusive and another branch, possibly off-diagonal can bifurcate from $\symbf{x}_1$. 

\subsubsection{Antisymmetric fixed points}
To see whether we have fixed points for the case $r_1\neq r_2$ we can consider the uncoupled case $\varepsilon = 0$



\begin{equation}
	\begin{bmatrix}
		r_1 & 0 \\
		0 & r_2
	\end{bmatrix}
	\begin{bmatrix}
		x(1-x)\\y(1-y)
	\end{bmatrix}
	-
	\begin{bmatrix}
		x\\y
	\end{bmatrix}
	=
	\begin{bmatrix}
		0\\0
	\end{bmatrix}
\end{equation}

Then, 
\begin{equation}
	\symbf{x}_2(r_1, r_2) = \left(1 - \frac{1}{r_1}, 1 - \frac{1}{r_2}\right)
\end{equation}
which is off-diagonal and away from the singularity $\varepsilon = 1/2$ found in the determinant in Equation (\ref{eq:singularityeq}). 
\subsubsection{Perturbation theory \& Antisymmetric fixed points}
Let
\[
r_1=r_0+\delta r_1^{(1)}+\mathcal O(\delta^2), \qquad
r_2=r_0+\delta r_2^{(1)}+\mathcal O(\delta^2),
\]
and seek a perturbative solutions of the form
\[
x(\delta)=x_0+\delta x_1+\mathcal O(\delta^2), \qquad
y(\delta)=y_0+\delta y_1+\mathcal O(\delta^2).
\]
to the fixed point equation.

Then
\begin{align}
x(1-x)
&=
\bigl(x_0+\delta x_1+\mathcal O(\delta^2)\bigr)
\Bigl(1-x_0-\delta x_1+\mathcal O(\delta^2)\Bigr) \notag\\
&=
x_0(1-x_0)+\delta x_1(1-2x_0)+\mathcal O(\delta^2),
\\[1em]
y(1-y)
&=
y_0(1-y_0)+\delta y_1(1-2y_0)+\mathcal O(\delta^2).
\end{align}

Now consider the fixed-point equations
\begin{align}
P(x,y) &= (1-\varepsilon)r_1 x(1-x)+\varepsilon r_2 y(1-y)-x = 0,\\
Q(x,y) &= (1-\varepsilon)r_2 y(1-y)+\varepsilon r_1 x(1-x)-y = 0.
\end{align}

Substituting the perturbation expansions gives
\begin{align*}
&(1-\varepsilon)\Bigl(r_0+\delta r_1^{(1)}\Bigr)
\Bigl(x_0(1-x_0)+\delta x_1(1-2x_0)\Bigr)
+\varepsilon\Bigl(r_0+\delta r_2^{(1)}\Bigr)
\Bigl(y_0(1-y_0)+\delta y_1(1-2y_0)\Bigr)
\notag\\
&\hspace{8cm}=
x_0+\delta x_1+\mathcal O(\delta^2),
\\[1em]
&(1-\varepsilon)\Bigl(r_0+\delta r_2^{(1)}\Bigr)
\Bigl(y_0(1-y_0)+\delta y_1(1-2y_0)\Bigr)
+\varepsilon\Bigl(r_0+\delta r_1^{(1)}\Bigr)
\Bigl(x_0(1-x_0)+\delta x_1(1-2x_0)\Bigr)
\notag\\
&\hspace{8cm}=
y_0+\delta y_1+\mathcal O(\delta^2).
\end{align*}

Collecting terms of order $\mathcal O(1)$ and $\mathcal O(\delta)$ yields the system
\begin{align*}
(1-\varepsilon)r_0\,x_0(1-x_0)+\varepsilon r_0\,y_0(1-y_0) &= x_0, \\
(1-\varepsilon)r_0\,y_0(1-y_0)+\varepsilon r_0\,x_0(1-x_0) &= y_0, \\
(1-\varepsilon)r_0\,x_1(1-2x_0)+\varepsilon r_0\,y_1(1-2y_0)
+(1-\varepsilon)r_1^{(1)}x_0(1-x_0)+\varepsilon r_2^{(1)}y_0(1-y_0) &= x_1, \\
(1-\varepsilon)r_0\,y_1(1-2y_0)+\varepsilon r_0\,x_1(1-2x_0)
+(1-\varepsilon)r_2^{(1)}y_0(1-y_0)+\varepsilon r_1^{(1)}x_0(1-x_0) &= y_1.
\end{align*}

For $\mathcal{O}(\delta)$ we have, $c_1 = (1-\varepsilon)r_0$, $c_2 = \varepsilon r_0$, $\sigma = r_1^{(1)}+r_2^{(1)}$, $t = 1 - 1/r_0$ and $\Delta = r_2^{(1)}- r_1^{(1)}$

\begin{equation}
	\begin{bmatrix}
		\alpha & \beta\\
		\beta & \alpha
	\end{bmatrix}
	\begin{bmatrix}
		x_1\\y_1
	\end{bmatrix}
	=
	\begin{bmatrix}
		c_1(1-2t) - 1 & c_2  (1-2t)\\
		c_2(1-2t) & c_1(1-2t) - 1
	\end{bmatrix}
	\begin{bmatrix}
	    x_1\\y_1
	\end{bmatrix}
	= 
	-u(t)
	\begin{bmatrix}
		r_1^{(1)} + \varepsilon \Delta\\
		r_2^{(1)} - \varepsilon \Delta
	\end{bmatrix}
	=
	\begin{bmatrix}
	    b_1\\b_2
	\end{bmatrix}
\end{equation}
\begin{equation}
	\begin{bmatrix}
		\alpha & \beta\\
		\beta & \alpha
	\end{bmatrix}
	\begin{bmatrix}
		x_1\\y_1
	\end{bmatrix}
	=
	VDV^{-1}
	\symbf{x}
	=
	\begin{bmatrix}
	    b_1\\b_2
	\end{bmatrix}
\end{equation}
\begin{equation}
	\symbf{x} = VD^{-1}V^{-1} \symbf{b} = VD^{-1}V^T\symbf{b}
\end{equation}
\begin{align}
	\lambda_\pm &= \alpha \pm \beta = (c_1 \pm c_2)(1-2t) - 1 \\
	\lambda_+ &= (c_1 + c_2)(1-2t) - 1= r_0(1-2t)-1  = (1-r_0)\\
	\lambda_- &= (c_1 - c_2)(1-2t) - 1= (1-2\varepsilon)r_0(1-2t)-1 = (1-2\varepsilon)(2-r_0) - 1\\
	\symbf{e}_\pm &= \frac{1}{\sqrt{2}}(\pm 1, 1)
\end{align}
This means that $a_-\symbf{e}_-$ forces $\symbf{x}$ away from the diagonal (smoothly).
\begin{equation}
	V =
	\frac{1}{\sqrt{2}}
	\begin{bmatrix}
		1 & -1\\
		1 & 1
	\end{bmatrix}
	,
	D^{-1} =
	\begin{bmatrix}
		\frac{1}{\lambda_+} & 0\\
		0 & \frac{1}{\lambda_-}
	\end{bmatrix}
\end{equation}
So 
\begin{equation}
	\symbf{x} =
		\frac{1}{2}
	\begin{bmatrix}
		1 & -1\\
		1 & 1
	\end{bmatrix}
	\begin{bmatrix}
		\frac{1}{\lambda_+} & 0\\
		0 & \frac{1}{\lambda_-}
	\end{bmatrix}
	\begin{bmatrix}
		1 & 1\\
		-1 & 1
	\end{bmatrix}
	\begin{bmatrix}
		b_1\\b_2
	\end{bmatrix}
\end{equation}
\begin{align}
	x_1 &= \frac{b_1+b_2}{2\lambda_+} - \frac{b_2-b_1}{2\lambda_-}\\
	y_1 &= \frac{b_1+b_2}{2\lambda_+} + \frac{b_2 - b_1}{2\lambda_-}
\end{align}
\begin{align}
	b_1 + b_2 &= -u(t)\sigma\\
	b_2 - b_1 &= -u(t)(1-2\varepsilon)\Delta
\end{align}
\begin{align}
	x_1 &= -\frac{1}{2\lambda_+}u(t)\sigma + \frac{1}{2\lambda_-} u(t)(1-2\varepsilon)\Delta =
	u(t)\left(-\frac{1}{2\lambda_+}\sigma + \frac{1}{2\lambda_-} (1-2\varepsilon)\Delta \right)\\
	y_1 &= -\frac{1}{2\lambda_+}u(t)\sigma - \frac{1}{2\lambda_-} u(t)(1-2\varepsilon)\Delta = 
	u(t)\left( -\frac{1}{2\lambda_+}\sigma - \frac{1}{2\lambda_-}(1-2\varepsilon)\Delta\right)
\end{align}
\begin{align}
	x_1 &= -\frac{u(t)}{2}\left(\frac{\sigma}{\lambda_+} + \frac{(1-2\varepsilon)\Delta}{\lambda_-}\right)\\
	y_1 &= -\frac{u(t)}{2}\left(\frac{\sigma}{\lambda_+} - \frac{(1-2\varepsilon)\Delta}{\lambda_-}\right)
\end{align}

\begin{equation}
	\symbf{x}_1 = -\frac{u(t)\sigma}{2\lambda_+} \symbf{e}_+ - \frac{u(t)(1-2\varepsilon)\Delta}{2\lambda_-}\symbf{e}_-
\end{equation}
Thus, the total perturbation of the diagonal fixed point is given by:

\begin{equation}
	\symbf{x}(\delta) = \symbf{x}_0 + \delta \symbf{x}_1 + \mathcal{O}(\delta^2) =  t\symbf{e}_+ + \delta \left(-\frac{u(t)\sigma}{2\lambda_+} \symbf{e}_+ - \frac{u(t)(1-2\varepsilon)\Delta}{2\lambda_-}\symbf{e}_-\right ) + \mathcal{O}(\delta^2)
\end{equation}

Which satisfies the fixed point equation up to first order as long as $\lambda_\pm \neq 0$ which are the same conditions the 
jacobian needs to satisfy to achieve invertibility on the diagonal fixed point.
