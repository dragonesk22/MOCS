  \caption{Two Gosper glider guns: the left gun provides a constant input stream, while the input channel injects gliders traveling in the south-east direction. No input is encoded by inserting an eater in the SE input stream. The output is read as gliders propagating in the south-west direction.}
  \label{fig:NOT_0}
\end{figure}

\begin{figure}[H]
  \centering
  \includegraphics[width=.75\textwidth]{figs/GoL/AND_IC}
  \caption{Three Gosper glider guns: the two leftmost generate south-east-directed input streams. An eater suppresses the south-west output unless both inputs
   are present. In the absence of inputs, additional eaters are placed along the collision paths to block spurious gliders. The output is read as a
   south-east-directed stream.}
  \label{fig:AND_0}
\end{figure}

\end{figure}

\subsubsection{NOT-gate}
\begin{figure}[H]
  \centering
  \includegraphics[width=\textwidth]{figs/GoL/NOT_488.pdf}
  \caption{One input channel, $A$. The input glider moves diagonally from the top to the lower-right part of the grid. The output is observed after the interaction region, in the outgoing diagonal channel. The construction is arranged so that when $A=0$ an output glider survives, while when $A=1$ the input glider destroys or blocks the outgoing signal.}
  \label{fig:NOT}
\end{figure}

\subsubsection{AND-gate}
\begin{figure}[H]
  \centering
  \includegraphics[width=\textwidth]{figs/GoL/AND_464.pdf}
  \caption{Two input channels are used, $A$ and $B$. The input gliders move diagonally from the top of the grid towards the lower-right part of the grid, where they interact with the gate structure. The output is observed in the outgoing diagonal channel after this interaction. A glider appears in the output channel only when both input gliders are
 present, so the output is $1$ only for $A=1$ and $B=1$.}
  \label{fig:AND}

\end{figure}
